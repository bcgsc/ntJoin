194 inline std::shared_ptr<cpptoml::table>
195 BloomFilter::parse_header(std::ifstream& file, const std::string& magic_string)
196 {
197  const std::string magic_with_brackets = std::string("[") + magic_string + "]";
198 
199  std::string line;
200  std::getline(file, line);
201  if (line != magic_with_brackets) {
202  log_error(
203  std::string("Magic string does not match (likely version mismatch)\n") +
204  "File magic string:\t" + line + "\n" + "Loader magic string:\t" +
205  magic_with_brackets);
206  std::exit(EXIT_FAILURE);
207  }
208 
209  /* Read bloom filter line by line until it sees "[HeaderEnd]"
210  which is used to mark the end of the header section and
211  assigns the header to a char array*/
212  std::string toml_buffer(line + '\n');
213  bool header_end_found = false;
214  while (bool(std::getline(file, line))) {
215  toml_buffer.append(line + '\n');
216  if (line == "[HeaderEnd]") {
217  header_end_found = true;
218  break;
219  }
220  }
221  if (!header_end_found) {
222  log_error("Pre-built bloom filter does not have the correct header end.");
223  std::exit(EXIT_FAILURE);
224  }
225 
226  // Send the char array to a stringstream for the cpptoml parser to parse
227  std::istringstream toml_stream(toml_buffer);
228  cpptoml::parser toml_parser(toml_stream);
229  const auto header_config = toml_parser.parse();
230 
231  // Obtain header values from toml parser and assign them to class members
232  return header_config->get_table(magic_string);
233 }
