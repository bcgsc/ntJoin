112 template<typename T>
113 inline void
114 CountingBloomFilter<T>::insert(const uint64_t* hashes)
115 {
116  // Update flag to track if increment is done on at least one counter
117  bool update_done = false;
118  T new_val;
119  T min_val = contains(hashes);
120  while (!update_done) {
121  // Simple check to deal with overflow
122  new_val = min_val + 1;
123  if (min_val > new_val) {
124  return;
125  }
126  for (size_t i = 0; i < hash_num; ++i) {
127  decltype(min_val) temp_min_val = min_val;
128  if (array[hashes[i] % array_size].compare_exchange_strong(temp_min_val,
129  new_val)) {
130  update_done = true;
131  }
132  }
133  // Recalculate minval because if increment fails, it needs a new minval to
134  // use and if it doesnt hava a new one, the while loop runs forever.
135  if (!update_done) {
136  min_val = contains(hashes);
137  }
138  }
139 }
